                        this.conn
                            .on('ready', () => {
                                console.log('----è¿æ¥ç›®æ ‡æœºæˆåŠŸ----')
                                resolve({
                                    success: true
                                })
                            })
                            .on('error', err => {
                                reject({
                                    success: false,
                                    error: err
                                })
                            })
                            .on('end', () => {
                                console.log('target ssh connect end!')
                            })

                                console.log('target ssh connect close')
                            })

                                sock: stream,

                reject({
                    success: false,
                    error: err
                })

                console.log('----SSHè¿æ¥å·²ç»“æŸ----')
            }).on('close', () => {
                console.log('----SSHè¿æ¥å·²å…³é—­----')
            }).connect(this.server)
        })
    }

    // ä¸Šä¼ æ–‡ä»¶
    uploadFile({ localPath, remotePath }) {
        return new Promise((resolve, reject) => {
            return this.conn.sftp((err, sftp) => {
                if (err) {
                    reject({
                        success: false,
                        error: err
                    })
                } else {
                    sftp.fastPut(localPath, remotePath, (err, result) => {
                        if (err) {
                            reject({
                                success: false,
                                error: err
                            })
                        }
                        resolve({
                            success: true,
                            result
                        })
                    })
                }
            })
        })
    }

    // æ‰§è¡Œsshå‘½ä»¤
    execSsh(command) {
        return new Promise((resolve, reject) => {
            return this.conn.exec(command, (err, stream) => {
                if (err || !stream) {
                    reject({
                        success: false,
                        error: err
                    })
                } else {
                    stream
                        .on('close', (code, signal) => {
                            resolve({
                                success: true
                            })
                        })
                        .on('data', function(data) {
                            console.log(data.toString())
                        })
                        .stderr.on('data', function(data) {
                            resolve({
                                success: false,
                                error: data.toString()
                            })
                        })
                }
            })
        })
    }

    // ç»“æŸè¿æ¥
    endConn() {
        this.conn.end()
        if (this.connAgent) {
            this.connAgent.end()
        }
    }
}
